Camera
by REAS <http://reas.com>

Reading and displaying an image from an attached camera

Updated 8 April 2005

    
// Camera 
// by REAS <http://reas.com> 

import processing.video.*; 
 
Camera c; 
 
void setup() 
{ 
  size(200, 200); 
 
  //printarr(Camera.list()); 
  String s = "Logitech QuickCam Messenger-WDM"; 
  c = new Camera(this, s, width, height, 30); 
  
  // If "null" is passed into the Camera constructor 
  // as the second parameter, a windows opens that 
  // lets you select camera options 
  //c = new Camera(this, null, 320, 240, 12); 
} 
 
void cameraEvent(Camera c) 
{ 
  c.read(); 
} 
 
void draw() 
{ 
  image(c, 0, 0); 
}
