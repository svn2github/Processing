Value Server
by hbarragan

Starts a network server in the local computer using the port 5204. It sends the connected clients values between 0 and 255. Click the mouse to stop the network server.

Created 23 June 2003

    
// Value Server 
// by hbarragan 

BFont metaBold; 
int port = 5204; 
boolean server_running = true; 
int bgColor = 0; 
int direction = 1; 
 
void setup() 
{ 
  size(200, 200); 
  background(0); 
  metaBold= loadFont("Meta-Bold.vlw.gz"); 
  textFont(metaBold, 44); 
  beginNet(port); // Starts a server on port 5204 
  framerate(25); 
} 
 
public void mouseClicked() 
{ 
  // If the mouse clicked the server stops 
  endNet(); 
  server_running = false; 
} 
 
void loop() 
{ 
  background(0); 
  if(server_running == true) 
  { 
    text("Server", 15, 60); 
    text("sending", 15, 95); 
    text("color "+bgColor, 15, 130); 
  } 
  else 
  { 
    text("server", 15, 60); 
    text("stopped", 15, 95); 
  } 
  netWrite(Integer.toString(bgColor)); 
  bgColor = bgColor + 1 * direction ; 
  if ((bgColor == 0) || (bgColor == 255)) { 
    direction *=-1; 
  } 
}
