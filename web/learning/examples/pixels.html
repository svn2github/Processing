Pixels
by hbarragan

Load a QuickTime movie file and display the video signal using rectangles as pixels by reading the values stored in the current video frame pixels array

Created 21 June 2003
Updated 3 August 2003

    
// Pixels 
// by hbarragan 

int numPixels; 
BVideo vid; 
 
void setup() 
{ 
  size(200, 200); 
  noStroke(); 
  background(0); 
  vid = loadVideo("station.mov"); 
  repeat(vid); 
  numPixels = width / 5; 
} 
 
void loop() 
{ 
  for(int j=0; j<numPixels; j=j+1) { 
    for(int i=0; i<numPixels; i=i+1) { 
      color pix = vid.pixels[j*vid.width+i]; 
      fill(red(pix), green(pix), blue(pix)); 
      rect(i*5, j*5, 4, 4); 
    } 
  } 
}
